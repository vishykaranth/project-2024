function replayEvents(streamId, fromVersion, toVersion):
    // 1. Load events from event store
    events = eventStore.readStream(streamId, fromVersion, toVersion)
    
    // 2. Initialize empty state
    state = createEmptyState()
    
    // 3. Apply each event in order
    for event in events:
        state = applyEvent(state, event)
    
    return state

function applyEvent(state, event):
    switch event.type:
        case "OrderCreated":
            state.orderId = event.payload.orderId
            state.status = "CREATED"
            state.total = 0
            return state
            
        case "ItemAdded":
            state.items.add(event.payload.item)
            state.total += event.payload.price
            return state
            
        case "PaymentProcessed":
            state.status = "PAID"
            state.paymentId = event.payload.paymentId
            return state
            
        case "OrderShipped":
            state.status = "SHIPPED"
            state.trackingNumber = event.payload.trackingNumber
            return state

function replayWithSnapshot(streamId, targetVersion):
    // 1. Find latest snapshot before target version
    snapshot = findLatestSnapshot(streamId, targetVersion)
    
    if snapshot exists:
        // 2. Start from snapshot state
        state = snapshot.state
        fromVersion = snapshot.version + 1
    else:
        // 3. No snapshot, start from beginning
        state = createEmptyState()
        fromVersion = 1
    
    // 4. Replay only events after snapshot
    events = eventStore.readStream(streamId, fromVersion, targetVersion)
    
    // 5. Apply remaining events
    for event in events:
        state = applyEvent(state, event)
    
    return state

Example:
    Target: Version 10,000
    Snapshot: Version 5,000
    Events to replay: 5,001 to 10,000 (5,000 events)
    Instead of: 1 to 10,000 (10,000 events)
    Speed improvement: 2x

### Aggregate with Event Sourcing
```java
public class OrderAggregate {
    private String orderId;
    private OrderStatus status;
    private BigDecimal total;
    private List<Event> uncommittedEvents = new ArrayList<>();
    private long version = 0;
    
    // Reconstruct from events
    public static OrderAggregate fromHistory(List<Event> events) {
        OrderAggregate aggregate = new OrderAggregate();
        for (Event event : events) {
            aggregate.apply(event);
        }
        aggregate.version = events.size();
        return aggregate;
    }
    
    // Apply event (mutate state)
    private void apply(Event event) {
        switch (event.getType()) {
            case "OrderCreated":
                OrderCreated e = (OrderCreated) event;
                this.orderId = e.getOrderId();
                this.status = OrderStatus.CREATED;
                this.total = BigDecimal.ZERO;
                break;
            case "ItemAdded":
                ItemAdded e2 = (ItemAdded) event;
                this.total = this.total.add(e2.getPrice());
                break;
            // ... other event types
        }
    }
    
    // Command: Ship order
    public void ship(String trackingNumber) {
        if (this.status != OrderStatus.PAID) {
            throw new IllegalStateException("Order must be paid");
        }
        
        OrderShipped event = new OrderShipped(this.orderId, trackingNumber);
        this.apply(event);
        this.uncommittedEvents.add(event);
    }
    
    public List<Event> getUncommittedEvents() {
        return new ArrayList<>(uncommittedEvents);
    }
    
    public void markEventsAsCommitted() {
        this.uncommittedEvents.clear();
        this.version += uncommittedEvents.size();
    }
}
```

## Key Takeaways

### Event Sourcing Core Concepts
```
1. Store events, not current state
2. Events are immutable and append-only
3. Replay events to reconstruct state
4. Use snapshots to optimize replay
5. Create projections for read models
6. Complete audit trail by default
```
